export interface PedagogicalImage {
  id: string;
  title: string;
  titleAr: string;
  description: string;
  imageUrl: string;
  thumbnailUrl: string;
  metadata: ImageMetadata;
  filterResults: FilterResults;
  status: 'pending' | 'approved' | 'rejected' | 'needs_review';
  uploadedAt: Date;
  reviewedAt?: Date;
}

export interface ImageMetadata {
  subject: Subject;
  educationLevel: EducationLevel;
  ageRange: { min: number; max: number };
  imageType: ImageType;
  cognitiveComplexity: 1|2|3|4|5;
  culturalContext: CulturalContext;
  tags: string[];
  concepts: string[];
}

export interface FilterResults {
  // الفلتر 1: الاستلاب الثقافي
  culturalFilter: {
    score: number;           // 0-100 (100 = آمن تماماً)
    issues: string[];
    suggestions: string[];
    passed: boolean;
  };
  // الفلتر 2: الواقعية الفائقة المعطِّلة
  realismFilter: {
    score: number;
    hypperealismLevel: 'low'|'medium'|'high'|'extreme';
    cognitiveBurdenRisk: boolean;
    imaginationInhibition: boolean;
    passed: boolean;
  };
  // الفلتر 3: المحاكاة الزائفة
  simulacrumFilter: {
    score: number;
    hasRealReference: boolean;
    semioticClarity: boolean;
    passed: boolean;
  };
  overallScore: number;
  overallPassed: boolean;
  aiAnalysis: string;   // تحليل Claude النصي
}

export type Subject =
  | 'arabic_language' | 'mathematics' | 'science' | 'history'
  | 'geography' | 'islamic_education' | 'french' | 'english'
  | 'art' | 'physical_education' | 'civic_education' | 'philosophy'

  | 'preschool'    // ما قبل المدرسة (3-5 سنوات)

        import Anthropic from '@anthropic-ai/sdk';

const client = new Anthropic({
  apiKey: import.meta.env.VITE_ANTHROPIC_API_KEY,
  dangerouslyAllowBrowser: true
});

// ===== الفلتر الأول: الاستلاب الثقافي =====
export async function runCulturalFilter(imageBase64, metadata) {
  const prompt = `أنت ناقد بصري متخصص في الأيقونوغرافيا التعليمية العربية.
حلّل الصورة وفق: الهوية البصرية، الشخصيات، البيئة المكانية، الانزياح التلقائي.
أجب بـ JSON: { "score": 0-100, "issues": [], "suggestions": [], "passed": bool }`;

  const response = await client.messages.create({
    model: 'claude-opus-4-5', max_tokens: 1000,
    messages: [{ role: 'user', content: [
      { type: 'image', source: { type: 'base64', media_type: 'image/jpeg', data: imageBase64 } },
      { type: 'text', text: prompt }
    ]}]
  });
  return JSON.parse(response.content[0].text.replace(/```json|```/g, '').trim());
}

// ===== الفلترة الشاملة (بالتوازي) =====
export async function runFullFilter(imageBase64, metadata) {
  const [cultural, realism, simulacrum] = await Promise.all([
    runCulturalFilter(imageBase64, metadata),
    runRealismFilter(imageBase64, metadata),
    runSimulacrumFilter(imageBase64, metadata)
  ]);

  const overallScore = Math.round(
    (cultural.score * 0.4) + (realism.score * 0.35) + (simulacrum.score * 0.25)
  );

  return { culturalFilter: cultural, realismFilter: realism,
           simulacrumFilter: simulacrum, overallScore,
           overallPassed: cultural.passed && realism.passed && simulacrum.passed };
}
